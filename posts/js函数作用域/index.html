闭包的引用：js的防抖和节流

防抖
function debounce(fun,delay){
	let timer
	return function(args){
		clearTimeout(timer)
		timer = setTimeout(function(){
			fun(args)
		},delay)
}
}

节流
function throottle(fun,wait){
	let timeout
	return function(){
		if(!timeout){
			timeout = setTimeout(function(){
				timeout=null
				fun
			},wait)
		}
	}
}
